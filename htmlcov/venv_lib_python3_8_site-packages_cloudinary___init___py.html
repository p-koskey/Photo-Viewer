108 def _put_nested_key(self, key, value): 

109 chain = re.split(r'[\[\]]+', key) 

110 chain = [k for k in chain if k] 

111 outer = self.__dict__ 

112 last_key = chain.pop() 

113 for inner_key in chain: 

114 if inner_key in outer: 

115 inner = outer[inner_key] 

116 else: 

117 inner = dict() 

118 outer[inner_key] = inner 

119 outer = inner 

120 if isinstance(value, list): 

121 value = value[0] 

122 outer[last_key] = value

331 def _generate_breakpoints(self, srcset_data): 

332 """ 

333 Helper function. Calculates static responsive breakpoints using provided parameters. 

334 

335 Either the breakpoints or min_width, max_width, max_images must be provided. 

336 

337 :param srcset_data: A dictionary containing the following keys: 

338 breakpoints A list of breakpoints. 

339 min_width Minimal width of the srcset images 

340 max_width Maximal width of the srcset images. 

341 max_images Number of srcset images to generate. 

342 

343 :return: A list of breakpoints 

344 

345 :raises ValueError: In case of invalid or missing parameters 

346 """ 

347 breakpoints = srcset_data.get("breakpoints", list()) 

348 

349 if breakpoints: 

350 return breakpoints 

351 

352 if not self._validate_srcset_data(srcset_data): 

353 return None 

354 

355 min_width, max_width, max_images = srcset_data["min_width"], srcset_data["max_width"], srcset_data["max_images"] 

356 

357 if max_images == 1: 

358 # if user requested only 1 image in srcset, we return max_width one 

359 min_width = max_width 

360 

361 step_size = int(ceil(float(max_width - min_width) / (max_images - 1 if max_images > 1 else 1))) 

362 

363 curr_breakpoint = min_width 

364 

365 while curr_breakpoint < max_width: 

366 breakpoints.append(curr_breakpoint) 

367 curr_breakpoint += step_size 

368 

369 breakpoints.append(max_width) 

370 

371 return breakpoints

692 def video(self, **options): 

693 """ 

694 Creates an HTML video tag for the provided +source+ 

695 

696 Examples: 

697 CloudinaryResource("mymovie.mp4").video() 

698 CloudinaryResource("mymovie.mp4").video(source_types = 'webm') 

699 CloudinaryResource("mymovie.ogv").video(poster = "myspecialplaceholder.jpg") 

700 CloudinaryResource("mymovie.webm").video(source_types = ['webm', 'mp4'], poster = {'effect': 'sepia'}) 

701 

702 :param options: 

703 * <tt>source_types</tt> - Specify which source type the tag should include. 

704 defaults to webm, mp4 and ogv. 

705 * <tt>sources</tt> - Similar to source_types, but may contain codecs list. 

706 source_types and sources are mutually exclusive, only one of 

707 them can be used. If both are not provided, default source types 

708 are used. 

709 * <tt>source_transformation</tt> - specific transformations to use 

710 for a specific source type. 

711 * <tt>poster</tt> - override default thumbnail: 

712 * url: provide an ad hoc url 

713 * options: with specific poster transformations and/or Cloudinary +:public_id+ 

714 

715 :return: Video tag 

716 """ 

717 public_id = options.get('public_id', self.public_id) 

718 source = re.sub(r"\.({0})$".format("|".join(self.default_source_types())), '', public_id) 

719 

720 custom_attributes = options.pop("attributes", dict()) 

721 

722 fallback = options.pop('fallback_content', '') 

723 

724 # Save source types for a single video source handling (it can be a single type) 

725 source_types = options.get('source_types', "") 

726 

727 poster_options = options.copy() 

728 if "poster" not in custom_attributes: 

729 options["poster"] = self._generate_video_poster_attr(source, poster_options) 

730 

731 if "resource_type" not in options: 

732 options["resource_type"] = self.resource_type or "video" 

733 

734 # populate video source tags 

735 source_tags = self._populate_video_source_tags(source, options) 

736 

737 if not source_tags: 

738 source = source + '.' + utils.build_array(source_types)[0] 

739 

740 video_url, video_options = utils.cloudinary_url(source, **options) 

741 

742 if not source_tags: 

743 custom_attributes['src'] = video_url 

744 

745 attributes = self._collect_video_tag_attributes(video_options) 

746 attributes.update(custom_attributes) 

747 

748 sources_str = ''.join(str(x) for x in source_tags) 

749 html = "<video {attributes}>{sources}{fallback}</video>".format( 

750 attributes=utils.html_attrs(attributes), sources=sources_str, fallback=fallback) 

751 

752 return html
